代码：爬豆瓣电影Top100

方式一： re + urllib

# _*_ coding:utf-8 _*_


'''
<span class="title">肖申克的救赎</span>
<em class="">1</em>
'''
import io
import sys
import re
from urllib.request import urlopen
sys.stdout = io.TextIOWrapper(sys.stdout.buffer,encoding='gb18030')
top_num = 1
url_base = 'https://movie.douban.com/top250?start='
urls = [url_base + str(i*25) for i in range(0,4)]
for url in urls:
    html = urlopen(url).read().decode('utf-8')
    top_tag = re.compile(r'<span class="title">(.*)</span>')
    title = re.findall(top_tag, html)
    for i in title:
        if i.find('/') == -1:
            print('Top' + str(top_num) + ' ' + i)
            top_num += 1

方式二： requests + BeautifulSoup

# _*_ coding:utf-8 _*_
'''
<span class="title">肖申克的救赎</span>
pip install requests
pip install BeautifulSoup
'''
import requests
from  bs4 import  BeautifulSoup
top_num = 1
url = 'https://movie.douban.com/top250?start='
urls = [url + str(i*25) for i in range(0,4)]
for url in urls:
    r = requests.get(url)
    html = r.text
    # 实例化
    soup = BeautifulSoup(html,'html.parser')
    movies = soup.find_all('span', class_='title')
    for i in movies:
        if i.text.find('/') == -1:
            print('Top' + str(top_num) + ' ' + i.text )
            top_num += 1
